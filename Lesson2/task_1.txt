var a = 1, b = 1, c, d;
c = ++a; alert(c);           // 2 |Инкремент ++ увеличивает переменную a на 1 ---> c=a+1; c=1+1=2 префиксная форма
d = b++; alert(d);           // 1 |Инкремент ++ увеличивает переменную b на 1 ---> d=b+1; значение переменной
                             увеличилось, но результат выдается до увеличения переменной равный d=b=1 постфиксная форма
c = (2+ ++a); alert(c);      // 5 |число 2 остается и не имеет приоритета, а = 2 из предыдущего расчета и ++ прибавляет 1
                                итого: 2+2+1=5
d = (2+ b++); alert(d);      // 4 |число 2 остается и не имеет приоритета, а = 2 из предыдущего расчета и ++ прибавляет 1,
                                  но не учиывает единицу итого: 2+2=4
alert(a);                    // 3 |из пред расчета сначала прибавили c = ++a, где a стало равно 2 далее прибавили
                                    c = (2+ ++a) ------>      итого а = 3
alert(b);                    // 3 |из пред расчета сначала прибавили d = b++, где b стало равно 2 далее прибавили
                                    d = (2+ b++) ------>      итого b = 3
                              независимо от того префиксная форма или постфиксная форма
